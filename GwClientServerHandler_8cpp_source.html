86  // Relay the payload

88  if (l_ServiceMessage.Deserialize(a_HigherLayerPayload)) {
89  // Check if it is directed to the address service

93  m_GwClient.Send(l_AddressAssignmentReply.Serialize());
94  } // if

96  return;
97  } // if
98 
99  // Check if it is directed to the publish / subscribe service

101  auto l_PublishSubscribeConfirmation = m_PublishSubscribeService.ProcessRequest(l_ServiceMessage, m_AddressLease);
102  if (l_PublishSubscribeConfirmation.GetDstServiceId() == 0xB0) {
103  m_GwClient.Send(l_PublishSubscribeConfirmation.Serialize());
104  } // if
105 
106  return;

108
