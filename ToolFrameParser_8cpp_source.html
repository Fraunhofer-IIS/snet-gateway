47 size_t ToolFrameParser::AddChunk(const unsigned char* a_Buffer, size_t a_Bytes) {
48  if (m_bStartTokenSeen == false) {
49  // No start token seen yet. Check if there is the start token available in the input buffer.
50  const void* l_pStartTokenPtr = memchr((const void*)a_Buffer, 0x7E, a_Bytes);
51  if (l_pStartTokenPtr) {
52  // The start token was found in the input buffer.
53  m_bStartTokenSeen = true;
54  if (l_pStartTokenPtr == a_Buffer) {
55  // The start token is at the beginning of the buffer. Clip it.
56  return 1;
57  } else {
58  // Clip front of buffer containing junk, including the start token.
59  return ((const unsigned char*)l_pStartTokenPtr - a_Buffer + 1);
60  } // else
61  } else {
62  // No token found, and no token was seen yet. Dropping received buffer now.
63  return a_Bytes;
64  } // else
65  } else {

67  const void* l_pEndTokenPtr = memchr((const void*)a_Buffer, 0x7E, a_Bytes);
